class DataromaScraper:
    """Scrapes superinvestor 13F holdings from Dataroma."""

    BASE_URL = "https://www.dataroma.com/m/holdings.php?m={manager_code}"

    def get_portfolio(self, manager_code: str) -> pd.DataFrame:
        url = self.BASE_URL.format(manager_code=manager_code)
        resp = self.session.get(url, timeout=15)

        # Parse each row for ticker, company, shares, value...

class ArkScraper:
    """Downloads ARK Invest holdings via arkfunds.io API."""

    ARK_API = "https://arkfunds.io/api/v2/etf/holdings?symbol={etf}"

    def get_holdings_api(self, etf: str) -> pd.DataFrame:
        url = self.ARK_API.format(etf=etf)
        resp = self.session.get(url, timeout=15)
        data = resp.json()
        holdings = data.get("holdings", [])
        df = pd.DataFrame(holdings)

def combine_dataroma(portfolios: dict) -> pd.DataFrame:
    """Stack all Dataroma portfolios into one DataFrame."""
    return pd.concat(portfolios.values(), ignore_index=True)

def top_conviction_picks(df, n=10) -> pd.DataFrame:
    """Find stocks held by the most managers, weighted by %."""
    agg = df.groupby("Symbol").agg(
        Stock      =("Stock", "first"),
        Managers   =("Manager", lambda x: ", ".join(x.unique())),
        Appearances=("Manager", "nunique"),
        Avg_Weight =("Weight", "mean"),
        Total_Value=("Value", "sum"),
    )
    return agg.sort_values(["Appearances"], ascending=False).head(n)

def save_results(df, filename: str):
    path = os.path.join("output", filename)
    os.makedirs("output", exist_ok=True)
    df.to_csv(path, index=False)
def main():
    # --- Dataroma ---
    dr = DataromaScraper()
    dr_portfolios = dr.get_all_portfolios()
    dr_combined = combine_dataroma(dr_portfolios)

    for code, name in DATAROMA_PORTFOLIOS.items():

    save_results(conviction, "dataroma_top_conviction.csv")

    # --- ARK ---
    ark = ArkScraper()
    ark_holdings = ark.get_all_holdings()
    for etf in ARK_ETFS:

            save_results(df, f"ark_{etf.lower()}_holdings.csv")

if __name__ == "__main__":
